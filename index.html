Moving Eyes with JavaScript

I have tried to make an eye which moves with the mouse using JavaScript.
This code uses the mouses relative location to the centre of the eye and then works how how far to move it.
First I made a simple eye with inkscape containing the layer containing the iris / pupil (which for me has an id 'layer1').
Then I opened the saved image with a text editor and copied it into the html where. Following the image in the code I inserted the JavaScript.
Fork

var movement = 20;
eye = document.getElementById('SVGRoot');
Iris = eye.getElementById("layer1");
document.onmousemove = function(e){
    let midX = (eye.getBoundingClientRect().left + eye.getBoundingClientRect().right)/2;
    let midY = (eye.getBoundingClientRect().top + eye.getBoundingClientRect().bottom)/2;
    if(event.clientX < midX){
        var x = movement* (event.clientX - midX) / midX + "%";
    }else if (event.clientX > midX){
        var x = movement* (event.clientX - midX) / (window.innerWidth - midX) + "%";
    }else{
        var x = "0%"
    };
    if(event.clientY < midY){
        var y = movement* (event.clientY - midY) / midY + "%";
    }else if(event.clientY > midY){
        var y = movement* (event.clientY - midY) / (window.innerHeight - midY) + "%";
    }else{
        var y = "0%"
    };
    Iris.style.left = x;
    Iris.style.top = y;
    Iris.style.transform = "translate("+ x +","+ y +")";
}
